// ==========================================
// 1. Volume Weighted Bands
// ==========================================
gp_vwb      = "Volume Weighted Bands"
visualMode  = input.string("LineOnly", "Mode", options = ["Cloud", "LineOnly"], inline='vwb', group=gp_vwb)
vwb_len     = input.int(24, "Trend", inline='vwb', group=gp_vwb)
smoothLen   = input.int(14, "Extra 평활", inline='vwb', group=gp_vwb)
bandMult    = input.float(1.0, "Band Width (ATR Multiplier)", step = 0.1, inline='vwb', group=gp_vwb)
atrLen      = input.int(14, "ATR Length", inline='vwb', group=gp_vwb)

float trend_Line = ta.ema((ta.vwma(close, vwb_len) + ta.ema(close, vwb_len)) / 2, smoothLen)
float atr        = ta.atr(atrLen)
float upperBand  = trend_Line + atr * bandMult
float lowerBand  = trend_Line - atr * bandMult

var int trend_State = 0
trend_State := close > upperBand ? 1 : close < lowerBand ? -1 : trend_State
color bull = color.rgb(92, 240, 215), color bear = color.rgb(179, 42, 195), color neutral = color.new(color.gray, 80)
color trendCol = trend_State == 1 ? bull : trend_State == -1 ? bear : neutral

bool isCloud  = visualMode == "Cloud"
pUpper = plot(isCloud ? upperBand : na, "Upper Band", color.new(trendCol, 60))
pLower = plot(isCloud ? lowerBand : na, "Lower Band", color.new(trendCol, 60))
fill(pUpper, pLower, isCloud ? color.new(trendCol, 85) : na)
plot(isCloud or visualMode == "LineOnly" ? trend_Line : na, "Trend Line", trendCol, 4)

// ==========================================
// 2. Keltner Hull Suite
// ==========================================
var string hull_settings = '════════ Hull 설정 ════════'
var string visual_settings = '════════ 시각화 설정 ════════'

tooltip_hull_period        = 'Hull 이동평균(HMA) 계산에 사용되는 캔들 수입니다.'
tooltip_keltner_period     = '켈트너 채널 계산을 위한 룩백 기간입니다. 변동성 측정 범위를 결정합니다.'
tooltip_keltner_multiplier = '동적 지지/저항 밴드를 생성하기 위해 적용되는 승수입니다.'
tooltip_price_source       = '모든 계산에 사용되는 가격 데이터 소스입니다.'
tooltip_preset             = '다양한 매매 스타일과 시장 조건에 최적화된 사전 설정 구성을 선택합니다.'
tooltip_bar_coloring       = '현재 추세 방향에 따라 차트의 캔들 색상을 변경할지 여부를 선택합니다.'
tooltip_color_preset       = '차트 테마에 최적화된 사전 정의된 색상 조합을 선택합니다.'

priceSource       = input.source(close, '', inline='kelt', group = hull_settings, tooltip = tooltip_price_source)
presetConfig      = input.string('Default', '프리셋 구성', options=['Default', 'Fast Response', 'Smooth Trend'], inline='kelt', group = hull_settings, tooltip = tooltip_preset)
hullPeriod        = input.int(14, 'Hull MA 기간', inline='kelt', group = hull_settings, tooltip = tooltip_hull_period)
keltnerPeriod     = input.int(20, 'Keltner 기간', inline='kelt', group = hull_settings, tooltip = tooltip_keltner_period)
keltnerMultiplier = input.float(2.0, 'Keltner 승수', step = 0.1, inline='kelt', group = hull_settings, tooltip = tooltip_keltner_multiplier)

if presetConfig == 'Fast Response'
    hullPeriod        := 10
    keltnerPeriod     := 14
    keltnerMultiplier := 1.5
else if presetConfig == 'Smooth Trend'
    hullPeriod        := 18
    keltnerPeriod     := 30
    keltnerMultiplier := 2.5

enableBarColoring = input.bool(true, '캔들색상', inline='kelt', group = visual_settings, tooltip = tooltip_bar_coloring)
colorPreset       = input.string('Custom', '색상', options=['Classic', 'Aqua', 'Cosmic', 'Ember', 'Neon', 'Custom'], inline='kelt', group = visual_settings, tooltip = tooltip_color_preset)
bullishColorInput = input.color(#00ffaa, '상승추세', inline='kelt', group = visual_settings)
bearishColorInput = input.color(#ff0000, '하락추세', inline='kelt', group = visual_settings)
lineTransparency  = input.int(30, '라인 투명도', minval = 0, maxval = 100, step = 5, inline='kelt', group = visual_settings)
fillTransparency  = input.int(60, '배경 투명도', minval = 0, maxval = 100, step = 5, inline='kelt', group = visual_settings)

[bullishColor, bearishColor] = switch colorPreset
    'Classic' => [#00ff00, #ff0000]
    'Aqua'    => [#00bfff, #ff8c00]
    'Cosmic'  => [#49ffce, #9932cc]
    'Ember'   => [#ff6600, #00cccc]
    'Neon'    => [#ffff00, #ff00ff]
    'Custom'  => [bullishColorInput, bearishColorInput]

hullMA       = ta.hma(priceSource, hullPeriod)
tr_kelt      = ta.tr(true)
smoothTR     = ta.ema(tr_kelt, keltnerPeriod)
doubleSmooth = ta.ema(smoothTR, keltnerPeriod) 
deviation    = doubleSmooth * keltnerMultiplier

upperBound = hullMA + deviation
lowerBound = hullMA - deviation

var float trendLine = hullMA
trendLine := nz(trendLine[1], hullMA)

if upperBound < trendLine
    trendLine := upperBound
if lowerBound > trendLine
    trendLine := lowerBound

trendUp   = trendLine > trendLine[1]
trendDown = trendLine < trendLine[1]

var color currentTrend = bullishColor
var int kelt_trendState = 0 

if trendUp
    currentTrend := bullishColor
    kelt_trendState   := 1
else if trendDown
    currentTrend := bearishColor
    kelt_trendState   := -1

bool kelt_isLong  = kelt_trendState == 1 and kelt_trendState[1] != 1
bool kelt_isShort = kelt_trendState == -1 and kelt_trendState[1] != -1

mainTrendLine    = trendLine
shiftedTrendLine = trendLine[2]

lineColor = color.new(currentTrend, lineTransparency)
fillColor = color.new(currentTrend, fillTransparency)

mainPlot    = plot(mainTrendLine, title = '메인 추세선', color = lineColor, linewidth = 1)
shiftedPlot = plot(shiftedTrendLine, title = '이동 추세선', color = lineColor, linewidth = 4)
fill(mainPlot, shiftedPlot, title = '추세 밴드 배경', color = fillColor)

barColor = enableBarColoring ? currentTrend : na
barcolor(barColor, title = '추세 기반 캔들 색상')


// ==========================================
// 3. High Volume Nodes (HVN)
// ==========================================
grp_hvn = 'High Volume Nodes (HVN)'
bool hvn_enable       = input.bool(true, "HVN:", inline='hvn', group=grp_hvn)
bool hvn_showOnlyMax  = input.bool(true, "only Long HVN", inline='hvn', group=grp_hvn)
int  hvn_numSlices    = input.int(100, "Slices 수", minval = 1, maxval = 200, inline='hvn', group=grp_hvn)
int  hvn_lookbackBars = input.int(200, "Bars", minval = 10, inline='hvn', group=grp_hvn)

float hvn_pocPrice = na   
bool  hvn_isLong   = false 
bool  hvn_isShort  = false 

var box[] hvn_Boxes = array.new_box()
var label[] hvn_Labels = array.new_label()

if hvn_enable
    if barstate.islast
        int hvn_actualLookback = math.min(hvn_lookbackBars, bar_index + 1)
        float hvn_hi = high[0]
        float hvn_lo = low[0]
        
        for i = 1 to hvn_actualLookback - 1
            if not na(high[i]) and high[i] > hvn_hi
                hvn_hi := high[i]
            if not na(low[i]) and low[i] < hvn_lo
                hvn_lo := low[i]
        
        if hvn_hi > hvn_lo 
            float hvn_priceStep = (hvn_hi - hvn_lo) / hvn_numSlices
            
            if array.size(hvn_Boxes) > 0
                for b in hvn_Boxes
                    box.delete(b)
                array.clear(hvn_Boxes)
            if array.size(hvn_Labels) > 0
                for l in hvn_Labels
                    label.delete(l)
                array.clear(hvn_Labels)
            
            float[] hvn_volumes = array.new_float(hvn_numSlices, 0.0)
            
            for i = 0 to hvn_actualLookback - 1
                float currHigh = high[i]
                float currLow  = low[i]
                float currVol  = nz(volume[i]) > 0 ? volume[i] : 1.0
                
                if na(currHigh) or na(currLow)
                    continue
                    
                for j = 0 to hvn_numSlices - 1
                    float sliceLow = hvn_lo + (j * hvn_priceStep)
                    float sliceHigh = sliceLow + hvn_priceStep
                    
                    if currLow <= sliceHigh and currHigh >= sliceLow
                        array.set(hvn_volumes, j, array.get(hvn_volumes, j) + currVol)

            float hvn_maxVol = array.max(hvn_volumes)
            int hvn_maxVolIndex = array.indexof(hvn_volumes, hvn_maxVol)
            
            if hvn_maxVolIndex >= 0
                hvn_pocPrice := hvn_lo + (hvn_maxVolIndex * hvn_priceStep) + (hvn_priceStep / 2)
            
            int hvn_leftBase = bar_index - hvn_actualLookback + 1
            int hvn_left = math.max(0, hvn_leftBase) 

            for j = 0 to hvn_numSlices - 1
                float vol = array.get(hvn_volumes, j)
                if vol > 0 and hvn_maxVol > 0
                    float sliceLow = hvn_lo + (j * hvn_priceStep)
                    float sliceHigh = sliceLow + hvn_priceStep
                    
                    int boxWidth = int((vol / hvn_maxVol) * (hvn_actualLookback * 0.99)) 
                    int hvn_right = hvn_left + math.max(1, boxWidth) 
                    
                    box b = box.new(hvn_left, sliceHigh, hvn_right, sliceLow, border_color = color.new(#2196f3, 50), bgcolor = color.new(#00bcd4, 85))
                    array.push(hvn_Boxes, b)
                    
                    float pctOfMax = (vol / hvn_maxVol) * 100
                    if pctOfMax >= 10 
                        bool shouldShow = not hvn_showOnlyMax or (vol == hvn_maxVol)
                        if shouldShow
                            label lbl = label.new(hvn_right, (sliceHigh + sliceLow) / 2, text = str.tostring(sliceLow, "#,###.#"), style = label.style_none, textcolor = color.white, size = size.normal)
                            array.push(hvn_Labels, lbl)


// ==========================================
// 4. Supply and Demand Zones (SDZ)
// ==========================================
grp_sdz    = "Supply and Demand Zones"
bool  sdz_enable     = input.bool(true, "Supply & Demand", inline='sdz', group=grp_sdz)
color sdz_col_supply = input.color(color.orange, "Supply (저항)", inline='sdz', group=grp_sdz)
color sdz_col_demand = input.color(#009fd4, "Demand (지지)", inline='sdz', group=grp_sdz)

var sdz_vol_arr    = array.new<float>()
var sdz_bear_boxes = array.new<box>()
var sdz_bull_boxes = array.new<box>()
var sdz_BullDelta  = array.new<float>()
var sdz_BearDelta  = array.new<float>()

var int sdz_count_bear = 0
var int sdz_count_bull = 0
var table sdz_dash = table.new(position.bottom_right, 10, 10, color.new(color.black, 100))

// 진입 신호 (초기화)
bool sdz_isLong  = false
bool sdz_isShort = false

// 리페인팅 및 에러 방지를 위해 ATR은 전역에서 계산
float sdz_atr = ta.atr(200) * 2

if sdz_enable
    sdz_vol_arr.push(volume)
    if sdz_vol_arr.size() > 1000
        sdz_vol_arr.shift()

    bool sdz_extra_vol = volume > sdz_vol_arr.avg()
    bool sdz_bear_candle = close < open 
    bool sdz_bull_candle = close > open

    // --- Supply 박스 생성 로직 ---
    if sdz_bear_candle and sdz_bear_candle[1] and sdz_bear_candle[2] and sdz_extra_vol[1] and sdz_count_bear == 0
        float sdz_delta = 0.
        for i = 0 to 5
            if sdz_bull_candle[i]
                sdz_count_bear := 1
                sdz_bear_boxes.push(box.new(bar_index-i, low[i]+sdz_atr, bar_index, low[i], bgcolor = color(na), border_color = sdz_col_supply, border_width = 2, text = "Supply: " + str.tostring(sdz_delta, format.volume), text_halign = text.align_right, text_size = size.normal))
                sdz_BearDelta.push(sdz_delta)
                break
            sdz_delta += sdz_bear_candle[i] ? -volume[i] : volume[i]

    if sdz_count_bear >= 1
        sdz_count_bear += 1
    if sdz_count_bear >= 15
        sdz_count_bear := 0

    // --- Demand 박스 생성 로직 ---
    if sdz_bull_candle and sdz_bull_candle[1] and sdz_bull_candle[2] and sdz_extra_vol[1] and sdz_count_bull == 0
        float sdz_delta = 0.
        for i = 0 to 5
            if sdz_bear_candle[i]
                sdz_count_bull := 1
                sdz_bull_boxes.push(box.new(bar_index-i, high[i], bar_index, high[i]-sdz_atr, bgcolor = color(na), border_color = sdz_col_demand, border_width = 2, text = "Demand: " + str.tostring(sdz_delta, format.volume), text_halign = text.align_right, text_size = size.normal))
                sdz_BullDelta.push(sdz_delta)
                break
            sdz_delta += sdz_bull_candle[i] ? volume[i] : -volume[i]

    if sdz_count_bull >= 1
        sdz_count_bull += 1
    if sdz_count_bull >= 15
        sdz_count_bull := 0

    // --- Supply 박스 관리 (역순 루프 적용) ---
    if sdz_bear_boxes.size() > 0
        for i = sdz_bear_boxes.size() - 1 to 0
            box b = sdz_bear_boxes.get(i)
            float sdz_buy = 0.
            float sdz_sell = 0.

            for bb in sdz_BullDelta
                sdz_buy += math.abs(bb)
            for ss in sdz_BearDelta
                sdz_sell += math.abs(ss)

            float sdz_tot = sdz_buy + sdz_sell

            b.set_text("Supply: " + str.tostring(sdz_BearDelta.get(i), format.volume) + " | " + str.tostring(math.abs(sdz_BearDelta.get(i) / sdz_tot * 100), format.percent))
            b.set_right(bar_index+30)   
            b.set_extend(extend.none)
            float sdz_top = b.get_top()
            float sdz_bot = b.get_bottom()
            b.set_bgcolor(color.new(sdz_col_supply, 90))

            // 시그널: 고가가 Supply Zone(저항) 안으로 들어올 때
            if high >= sdz_bot and high[1] < sdz_bot
                sdz_isShort := true

            if close > sdz_top 
                b.delete()
                sdz_bear_boxes.remove(i)
                sdz_BearDelta.remove(i)
                continue

            if (b.get_right() - b.get_left() - 15) > 20
                if high > sdz_bot and low < sdz_bot 
                    b.set_border_style(line.style_dashed)
                    b.set_border_width(1)

    // Supply 박스 중복 제거 (역순 루프 적용)
    if sdz_bear_boxes.size() > 0
        for i = sdz_bear_boxes.size() - 1 to 0
            box lvl = sdz_bear_boxes.get(i)
            float sdz_top = lvl.get_top()
            float sdz_bot = lvl.get_bottom()

            for j = 0 to sdz_bear_boxes.size() - 1
                if i == j
                    continue  
                box l = sdz_bear_boxes.get(j)
                float sdz_top1 = l.get_top()

                if sdz_top1 < sdz_top and sdz_top1 > sdz_bot
                    lvl.delete()
                    sdz_bear_boxes.remove(i)
                    sdz_BearDelta.remove(i)
                    break

    if sdz_bear_boxes.size() > 5 
        sdz_bear_boxes.shift().delete()
        sdz_BearDelta.shift()

    // --- Demand 박스 관리 (역순 루프 적용) ---
    if sdz_bull_boxes.size() > 0
        for i = sdz_bull_boxes.size() - 1 to 0
            box b = sdz_bull_boxes.get(i)
            float sdz_buy = 0.
            float sdz_sell = 0.

            for bb in sdz_BullDelta
                sdz_buy += math.abs(bb)
            for ss in sdz_BearDelta
                sdz_sell += math.abs(ss)

            float sdz_tot = sdz_buy + sdz_sell

            b.set_text("Demand: " + str.tostring(sdz_BullDelta.get(i), format.volume) + " | " + str.tostring(sdz_BullDelta.get(i) / sdz_tot * 100, format.percent))
            b.set_right(bar_index+30)   
            b.set_extend(extend.none)
            float sdz_bot = b.get_bottom()
            float sdz_top = b.get_top()
            b.set_bgcolor(color.new(sdz_col_demand, 90))

            // 시그널: 저가가 Demand Zone(지지) 안으로 들어올 때
            if low <= sdz_top and low[1] > sdz_top
                sdz_isLong := true

            if close < sdz_bot 
                b.delete()
                sdz_bull_boxes.remove(i)
                sdz_BullDelta.remove(i)
                continue

            if (b.get_right() - b.get_left() - 15) > 20
                if low < sdz_top and high > sdz_top 
                    b.set_border_style(line.style_dashed)
                    b.set_border_width(1)

    // Demand 박스 중복 제거 (역순 루프 적용)
    if sdz_bull_boxes.size() > 0
        for i = sdz_bull_boxes.size() - 1 to 0
            box lvl = sdz_bull_boxes.get(i)
            float sdz_top = lvl.get_top()
            float sdz_bot = lvl.get_bottom()

            for j = 0 to sdz_bull_boxes.size() - 1
                if i == j
                    continue  
                box l = sdz_bull_boxes.get(j)
                float sdz_bot1 = l.get_bottom()

                if sdz_bot1 < sdz_top and sdz_bot1 > sdz_bot
                    lvl.delete()
                    sdz_bull_boxes.remove(i)
                    sdz_BullDelta.remove(i)
                    break

    if sdz_bull_boxes.size() > 5 
        sdz_bull_boxes.shift().delete()
        sdz_BullDelta.shift()

    // --- 대시보드 출력 ---
    if barstate.islast
        sdz_dash.cell(0, 0, "Total Supply:", text_halign = text.align_left, text_color = sdz_col_supply)
        sdz_dash.cell(0, 1, "Total Demand:", text_halign = text.align_left, text_color = sdz_col_demand)
        sdz_dash.cell(1, 0, str.tostring(sdz_BearDelta.sum()), text_color = sdz_col_supply)
        sdz_dash.cell(1, 1, str.tostring(sdz_BullDelta.sum()), text_color = sdz_col_demand)

alertcondition(kelt_isShort, title = '숏(하락) 전환 신호', message = 'Keltner Hull Suite: 하락 추세 전환 [{{exchange}}:{{ticker}}]')
